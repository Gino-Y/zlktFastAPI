一、返回数据的几种方式

import uvicorn

from fastapi import FastAPI,Response
from fastapi.responses import JSONResponse

app = FastAPI(
title="响应体测试",
    description="响应体测试",
)

# 1.返回字符串
@app.get("/ret_str")
def ret_str():
    return "hello fastapi"


# 2.返回字典
@app.get("/ret_dict")
def ret_dict():
    return {"id":1,"name":"zhiliao"}

# 3.返回json
@app.get("/ret_json")
def ret_json():
    # content = {"id":1,"name":"zhiliao"}
    # content = '{"id":1,"name":"zhiliao"}'
    content = """{"id":1,"name":"zhiliao"}"""
    return JSONResponse(content=content)

# 4.设置header:第一种方式,header的value只能是英文的，中文会出现编码错误
@app.get("/set_header")
def set_header(rsp:Response):
    rsp.headers["name"] = "hallen"
    return "zhiliao"

# 5.设置header:第二种方式，header的value只能是英文的，中文会出现编码错误
@app.get("/set_header2")
def set_header2():
    content = {"age":18}
    headers = {"name":"zhiliao","addr":"cs"}
    return JSONResponse(content=content,headers=headers)

# 6.自定义返回信息
@app.get("/ret_xml")
def ret_xml():
    content = """
    <?xml version="1.0">
    <user>
        <name>hallen</name>
        <age>18</age>
        <addr>长沙</addr>
    </user>
    """

    return Response(content=content,media_type="application/xml")
    # return content

if __name__ == '__main__':
    # 容易忽略py文件名称
    uvicorn.run(app="test04:app",reload=True,host="0.0.0.0",port=8080)

二、响应模型

前端发起请求，后端返回响应数据，这个数据可以是我们自己定义好的，为了省事，这里引入响应模型，主要是对返回数据的封装

  • 1.定义一个pydantic模型，作为返回的数据模型
from typing import List
from pydantic import BaseModel

class Article(BaseModel):
    id: int
    name: str
    tags:List[str] = []
  • 2.返回该模型数据
@app.get("/article/", response_model=Article)
def get_article():
    article = Article(id=2,title="文章测试")
    return article
  • 注意：
    • response_model不是路径参数，只是一个函数的参数
    • 从输出中移除一些数据，response_model返回的模型和return的可以不是同一个，比如模型上字段比较多，但是想返回其中的几个，可以再定义一个返回的模型，字段都可以在原来多的那个模型中找到，response_model这个字段少的模型，这样返回的就是字段少的
# 7.返回模型
class Article(BaseModel):
    id: int
    title: str

class User(BaseModel):
    id: int
    name: str
    pwd: str

class RetUser(BaseModel):
    id: int
    name: str

@app.get("/ret_model",response_model=RetUser)
def ret_model_data():
    # 假如数据从数据库查询出来
    user = User(id=1,name="hallen",pwd="123456")
    return user


return和response_model返回的不是同一个，但是其中有几个字段是相同的

response_model_include 或 response_model_exclude也可以实现这样的效果，排除掉几个字段，不建议使用

get请求或post等请求的函数中设置参数：
    response_model_exclude={"tax"}    # ["tax"]也可以，会转为set
    response_model_include={"name", "description"}
  • 3.response_model_exclude_unset=True参数

有时候数据会有默认值，比如数据库中设置了默认值，不想返回这些默认值怎么办？

response_model_exclude_unset=True参数设置后就不会返回默认值，只会返回实际设置的值

  • 4.返回多个模型
Union：
    from typing import Union
    from pydantic import BaseModel
    class Article(BaseModel):
        id: int
        title: str

    class User(BaseModel):
        id: int
        name: str
        pwd: str

    class RetUser(BaseModel):
        id: int
        name: str

    # Union 是or的意思
    @app.get("/ret_model",response_model=Union[RetUser,Article])  # List[User or Atticle]
    def ret_model_data():
        # 假如数据从数据库查询出来
        article = Article(id=2,title="文章测试")
        user = User(id=1,name="hallen",pwd="123456")
        return article

或者list：
      response_model=List[RetUser or Article]
  • 5.不使用模型
# 1.返回字典
@app.get("/keyword-weights/", response_model=Dict[str, float])
def get_info():
    return {"foo": 2.3, "bar": 3.4}


# 2.返回json
from fastapi.responses import JSONResponse
from fastapi.encoders import jsonable_encoder
@app.get("/keyword-weights/", response_model=Dict[str, float])
def get_info():
    item = {"foo": 2.3, "bar": 3.4}
    content = jsonable_encoder(item)   // 转为json返回，或者直接返回字典也可以
    return JSONResponse(status_code=status.HTTP_201_CREATED, content=content)
    # 也可以设置响应头
    # return JSONResponse(status_code=status.HTTP_201_CREATED, content=content,headers="xxx")


# 3.自定义返回，如果你直接返回 Response，返回数据不会自动转换，也不会自动生成文档
from fastapi import FastAPI, Response,status
@app.get("/legacy/")
def get_legacy_data():
    item = {"foo": 2.3, "bar": 3.4}
    return Response(content=item, media_type="application/xml")

三、状态码

status_code不是路径参数，只是一个函数的参数

1.自定义状态码
    @app.post("/user/", status_code=200)

2.使用常量
    @app.post("/user/", status_code=status.HTTP_200_OK)

user_in = UserIn(**user_dict)

五、装饰器的参数总结

1.状态码：status_code

@app.post("/items/",status_code=200)   


# 也可以使用状态码常量
from fastapi import FastAPI,Response,status

status.HTTP_200_OK

2.响应模型：response_model

@app.post("/items/",response_model=Article)

3.标签：tags，会在接口文档中体现

@app.get("/items/", tags=["items"])

4.简述：summary

@app.get("/items/", summary="Create an item")

5.详细描述：description

@app.get("/items/", description="Create an item with all the information, name, description, price, tax and a set of unique tags")

6.响应说明：response_description

@app.get("/items/", response_description="The created item")

7.弃用路径：deprecated=True，在api文档中将显示弃用

@app.get("/items/", deprecated=True)
